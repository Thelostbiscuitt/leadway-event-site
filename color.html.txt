    <script>
        // Persistent data storage using localStorage
        let posts = JSON.parse(localStorage.getItem('eventSharePosts')) || [];
        let currentTheme = localStorage.getItem('eventShareTheme') || 'light';
        let isCollageUnlocked = false;
        const COLLAGE_PASSWORD = "244466666";
        let selectedFiles = [];
        
        // Initialize the application
        document.addEventListener('DOMContentLoaded', function() {
            // Load saved theme
            setTheme(currentTheme);
            
            // Render saved posts
            renderPosts();
            
            // Initialize theme toggle text
            updateThemeToggleText();
            
            // Load stats
            updateStats();
        });
        
        // Toggle sidebar
        function toggleSidebar() {
            const sidebar = document.getElementById('sidebar');
            const overlay = document.getElementById('overlay');
            
            sidebar.classList.toggle('open');
            overlay.classList.toggle('open');
        }
        
        // Show specific page
        function showPage(pageId) {
            // Hide all pages
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => page.classList.remove('active'));
            
            // Show selected page
            const selectedPage = document.getElementById(pageId + 'Page');
            selectedPage.classList.add('active');
            
            // Update menu links
            document.querySelectorAll('.sidebar-menu a').forEach(link => link.classList.remove('active'));
            document.getElementById(pageId + 'Link').classList.add('active');
            
            // Update stats if needed
            if (pageId === 'stats') {
                updateStats();
            }
        }
        
        // Update statistics
        function updateStats() {
            const postCount = posts.length;
            const commentCount = posts.reduce((total, post) => total + post.comments.length, 0);
            const mediaCount = posts.reduce((total, post) => total + post.media.length, 0);
            const likeCount = posts.reduce((total, post) => total + post.likes, 0);
            
            document.getElementById('postCount').textContent = postCount;
            document.getElementById('commentCount').textContent = commentCount;
            document.getElementById('mediaCount').textContent = mediaCount;
            document.getElementById('likeCount').textContent = likeCount;
        }
        
        // Toggle theme
        function toggleTheme() {
            currentTheme = currentTheme === 'light' ? 'dark' : 'light';
            setTheme(currentTheme);
            localStorage.setItem('eventShareTheme', currentTheme);
            updateThemeToggleText();
        }
        
        // Apply theme to the document
        function setTheme(theme) {
            document.documentElement.setAttribute('data-theme', theme);
        }
        
        // Update theme toggle button text
        function updateThemeToggleText() {
            const icon = document.getElementById('themeIcon');
            const text = document.getElementById('themeText');
            
            if (currentTheme === 'light') {
                icon.textContent = 'ðŸŒ™';
                text.textContent = 'Dark Mode';
            } else {
                icon.textContent = 'â˜€ï¸';
                text.textContent = 'Light Mode';
            }
        }
        
        // Handle file selection
        function handleFiles() {
            const input = document.getElementById('mediaInput');
            const counter = document.getElementById('fileCounter');
            const fileList = document.getElementById('fileList');
            
            // Clear previous selection
            selectedFiles = [];
            fileList.innerHTML = '';
            
            // Get files and limit to 5
            const files = Array.from(input.files).slice(0, 5);
            selectedFiles = files;
            
            // Update counter
            counter.textContent = `${files.length}/5`;
            
            // Display file list
            if (files.length > 0) {
                files.forEach((file, index) => {
                    const fileItem = document.createElement('div');
                    fileItem.className = 'file-item';
                    fileItem.innerHTML = `
                        <div class="file-name">${file.name}</div>
                        <button class="remove-file" onclick="removeFile(${index})">Ã—</button>
                    `;
                    fileList.appendChild(fileItem);
                });
            }
        }
        
        // Remove a file from selection
        function removeFile(index) {
            selectedFiles.splice(index, 1);
            handleFiles(); // Refresh file display
        }
        
        // Create a new post
        function createPost() {
            const authorName = document.getElementById('authorName').value.trim();
            const postContent = document.getElementById('postContent').value.trim();
            
            if (!authorName || !postContent) {
                showNotification('Please enter your name and post content', 'error');
                return;
            }
            
            const newPost = {
                id: Date.now(),
                author: authorName,
                content: postContent,
                timestamp: new Date().toISOString(),
                likes: 0,
                liked: false,
                comments: [],
                media: []
            };
            
            // Handle media files
            if (selectedFiles.length > 0) {
                for (let i = 0; i < selectedFiles.length; i++) {
                    const file = selectedFiles[i];
                    const fileName = `media-${Date.now()}-${file.name}`;
                    const fileType = file.type.split('/')[0]; // 'image' or 'video'
                    
                    newPost.media.push({
                        name: fileName,
                        type: fileType,
                        url: URL.createObjectURL(file)
                    });
                }
            }
            
            // Add to posts array
            posts.unshift(newPost);
            
            // Save to localStorage
            savePosts();
            
            // Render posts
            renderPosts();
            
            // Clear form
            document.getElementById('authorName').value = '';
            document.getElementById('postContent').value = '';
            document.getElementById('mediaInput').value = '';
            document.getElementById('fileList').innerHTML = '';
            document.getElementById('fileCounter').textContent = '0/5';
            selectedFiles = [];
            
            showNotification('Post created successfully!');
        }
        
        // Render all posts
        function renderPosts() {
            const postsContainer = document.getElementById('postsContainer');
            
            if (posts.length === 0) {
                postsContainer.innerHTML = `
                    <div class="post" style="text-align: center; padding: 40px;">
                        <h3>No posts yet</h3>
                        <p>Be the first to share an event!</p>
                    </div>
                `;
                return;
            }
            
            postsContainer.innerHTML = '';
            
            posts.forEach(post => {
                const postElement = document.createElement('div');
                postElement.className = 'post';
                postElement.innerHTML = `
                    <div class="post-header">
                        <div class="avatar">${post.author.charAt(0)}</div>
                        <div class="post-info">
                            <h4>${post.author}</h4>
                            <div class="post-time">${formatDate(post.timestamp)}</div>
                        </div>
                    </div>
                    <div class="post-content">${post.content}</div>
                    ${post.media.length > 0 ? `
                        <div class="post-media">
                            ${post.media.map(media => 
                                media.type === 'image' ? 
                                    `<div class="media-item"><img src="${media.url}" alt="Post image"></div>` :
                                    `<div class="media-item"><video controls><source src="${media.url}" type="video/mp4"></video></div>`
                            ).join('')}
                        </div>
                    ` : ''}
                    <div class="post-actions">
                        <button class="action-btn ${post.liked ? 'liked' : ''}" onclick="toggleLike(${post.id})">
                            <i class="fas fa-heart"></i> <span>${post.likes}</span>
                        </button>
                        <button class="action-btn" onclick="toggleComments(this)">
                            <i class="fas fa-comment"></i> ${post.comments.length} Comments
                        </button>
                    </div>
                    <div class="comments-section" style="display: none;">
                        ${post.comments.length > 0 ? post.comments.map(comment => `
                            <div class="comment">
                                <div class="comment-header">
                                    <span class="comment-author">${comment.author}</span>
                                    <span class="comment-time">${formatDate(comment.timestamp)}</span>
                                </div>
                                <div>${comment.content}</div>
                            </div>
                        `).join('') : '<p style="text-align:center;color:var(--text-secondary)">No comments yet</p>'}
                        <div class="comment-form">
                            <input type="text" class="comment-author-input" placeholder="Your name">
                            <input type="text" class="comment-input" placeholder="Add a comment...">
                            <button class="comment-btn" onclick="addComment(${post.id}, this)">Post</button>
                        </div>
                    </div>
                `;
                
                postsContainer.appendChild(postElement);
            });
            
            updateStats();
        }
        
        // Toggle like on a post
        function toggleLike(postId) {
            const post = posts.find(p => p.id === postId);
            if (!post) return;
            
            post.liked = !post.liked;
            post.likes += post.liked ? 1 : -1;
            
            savePosts();
            renderPosts();
        }
        
        // Toggle comments visibility
        function toggleComments(button) {
            const commentsSection = button.closest('.post').querySelector('.comments-section');
            commentsSection.style.display = commentsSection.style.display === 'none' ? 'block' : 'none';
        }
        
        // Add comment to a post
        function addComment(postId, button) {
            const form = button.closest('.comment-form');
            const authorInput = form.querySelector('.comment-author-input');
            const commentInput = form.querySelector('.comment-input');
            
            const author = authorInput.value.trim();
            const commentText = commentInput.value.trim();
            
            if (!author || !commentText) {
                showNotification('Please enter your name and comment', 'error');
                return;
            }
            
            const post = posts.find(p => p.id === postId);
            if (!post) return;
            
            post.comments.push({
                author: author,
                content: commentText,
                timestamp: new Date().toISOString()
            });
            
            savePosts();
            renderPosts();
            
            // Clear inputs
            authorInput.value = '';
            commentInput.value = '';
        }
        
        // Check collage password
        function checkPassword() {
            const passwordInput = document.getElementById('passwordInput').value;
            const passwordError = document.getElementById('passwordError');
            
            if (passwordInput === COLLAGE_PASSWORD) {
                isCollageUnlocked = true;
                document.getElementById('passwordOverlay').style.display = 'none';
                showNotification('Collage generator unlocked!');
            } else {
                passwordError.style.display = 'block';
                setTimeout(() => {
                    passwordError.style.display = 'none';
                }, 3000);
            }
        }
        
        // Generate collage (simulated)
        function generateCollage(type) {
            if (!isCollageUnlocked) {
                showNotification('Please unlock the collage generator first', 'error');
                return;
            }
            
            // Get all media from posts
            const mediaItems = posts.flatMap(post => post.media);
            
            if (mediaItems.length === 0) {
                showNotification('No media available to create collage', 'error');
                return;
            }
            
            const canvas = document.getElementById('collageCanvas');
            const placeholder = document.getElementById('collagePlaceholder');
            
            // Clear previous collage
            canvas.innerHTML = '';
            
            // Show loading
            const loading = document.createElement('div');
            loading.className = 'loading';
            canvas.appendChild(loading);
            placeholder.style.display = 'none';
            
            // Simulate AI processing
            setTimeout(() => {
                canvas.innerHTML = '';
                placeholder.style.display = 'none';
                
                const collageImg = document.createElement('div');
                collageImg.style.width = '100%';
                collageImg.style.height = '100%';
                collageImg.style.background = 'linear-gradient(135deg, #6a11cb 0%, #2575fc 100%)';
                collageImg.style.borderRadius = '10px';
                collageImg.style.display = 'flex';
                collageImg.style.flexDirection = 'column';
                collageImg.style.alignItems = 'center';
                collageImg.style.justifyContent = 'center';
                collageImg.style.color = 'white';
                collageImg.style.fontSize = '1.5rem';
                collageImg.style.fontWeight = 'bold';
                collageImg.style.textAlign = 'center';
                collageImg.style.padding = '20px';
                
                collageImg.innerHTML = `
                    <div>
                        <div style="font-size: 3rem; margin-bottom: 20px;"><i class="fas fa-check-circle"></i></div>
                        ${type.charAt(0).toUpperCase() + type.slice(1)} Collage Created!
                        <div style="font-size: 1rem; margin-top: 20px;">
                            Generated from ${mediaItems.length} media items
                        </div>
                        <div style="font-size: 0.9rem; margin-top: 10px; opacity: 0.8;">
                            (Simulated AI-generated collage)
                        </div>
                    </div>
                `;
                
                canvas.appendChild(collageImg);
                
                showNotification(`${type} collage generated successfully!`);
            }, 2000);
        }
        
        // Clear collage
        function clearCollage() {
            const canvas = document.getElementById('collageCanvas');
            const placeholder = document.getElementById('collagePlaceholder');
            canvas.innerHTML = '';
            placeholder.style.display = 'block';
        }
        
        // Save posts to localStorage
        function savePosts() {
            localStorage.setItem('eventSharePosts', JSON.stringify(posts));
        }
        
        // Format date for display
        function formatDate(isoString) {
            const date = new Date(isoString);
            return date.toLocaleString('en-US', {
                month: 'short',
                day: 'numeric',
                year: 'numeric',
                hour: '2-digit',
                minute: '2-digit'
            });
        }
        
        // Show notification
        function showNotification(message, type = 'success') {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.className = `notification ${type} show`;
            
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }
